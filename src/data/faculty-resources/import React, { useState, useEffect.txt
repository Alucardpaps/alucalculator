const apiKey = ""; // API Anahtarınızı buraya girebilirsiniz.

const callGemini = async (prompt, systemInstruction = "") => {
  try {
    // Gerçek API çağrısı için bu kısmı kullanın.
    // Şimdilik demo modunda çalışır.
    if (!apiKey) return "API Anahtarı eksik. Lütfen koda anahtarınızı ekleyin.";
    
    const response = await fetch(
      `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`,
      {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          contents: [{ parts: [{ text: prompt }] }],
          systemInstruction: { parts: [{ text: systemInstruction }] }
        })
      }
    );

    if (!response.ok) throw new Error('API Error');
    const data = await response.json();
    return data.candidates?.[0]?.content?.parts?.[0]?.text || "Bir hata oluştu.";
  } catch (error) {
    console.error("Gemini Error:", error);
    return "Bağlantı hatası. Lütfen internetinizi veya API anahtarınızı kontrol edin.";
  }
};

// --- MODULE: AI Assistant ---
const AIAssistant = () => {
    const [messages, setMessages] = useState([
        { role: 'model', text: 'Merhaba! Ben Sanal Baş Mühendisiniz. Malzeme seçimi, formüller veya tasarım problemleri hakkında bana soru sorabilirsiniz.' }
    ]);
    const [input, setInput] = useState('');
    const [loading, setLoading] = useState(false);
    const messagesEndRef = useRef(null);

    const scrollToBottom = () => { messagesEndRef.current?.scrollIntoView({ behavior: "smooth" }); };
    useEffect(() => { scrollToBottom(); }, [messages]);

    const handleSend = async () => {
        if (!input.trim()) return;
        const userMsg = input;
        setMessages(prev => [...prev, { role: 'user', text: userMsg }]);
        setInput('');
        setLoading(true);

        const systemPrompt = "Sen uzman bir makine mühendisisin. Türk mühendislik terminolojisine hakimsin. Teknik sorulara kısa, net ve bilimsel yanıtlar ver.";
        const response = await callGemini(userMsg, systemPrompt);

        setMessages(prev => [...prev, { role: 'model', text: response }]);
        setLoading(false);
    };

    return (
        <div className="p-6 h-full max-w-4xl mx-auto flex flex-col">
            <div className="bg-white rounded-xl shadow-sm border border-slate-200 flex-1 flex flex-col overflow-hidden">
                <div className="p-4 border-b border-slate-100 bg-slate-50 flex items-center gap-2">
                    <Sparkles className="text-purple-600" size={20} />
                    <h2 className="font-bold text-slate-800">AI Baş Mühendis</h2>
                </div>
                <div className="flex-1 overflow-y-auto p-4 space-y-4">
                    {messages.map((msg, idx) => (
                        <div key={idx} className={`flex ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}>
                            <div className={`max-w-[80%] rounded-2xl px-4 py-3 text-sm leading-relaxed ${msg.role === 'user' ? 'bg-blue-600 text-white rounded-tr-none' : 'bg-slate-100 text-slate-800 rounded-tl-none'}`}>
                                {msg.text.split('\n').map((line, i) => <p key={i} className="min-h-[1rem]">{line}</p>)}
                            </div>
                        </div>
                    ))}
                    {loading && <div className="text-xs text-slate-500 p-2">Hesaplanıyor...</div>}
                    <div ref={messagesEndRef} />
                </div>
                <div className="p-4 border-t border-slate-100 bg-white flex gap-2">
                    <input type="text" value={input} onChange={(e) => setInput(e.target.value)} onKeyPress={(e) => e.key === 'Enter' && handleSend()} placeholder="Bir soru sorun..." className="flex-1 p-3 bg-slate-50 border border-slate-200 rounded-xl focus:outline-none focus:border-blue-500" />
                    <button onClick={handleSend} disabled={loading} className="p-3 bg-blue-600 text-white rounded-xl hover:bg-blue-700 disabled:opacity-50"><Send size={20} /></button>
                </div>
            </div>
        </div>
    );
};

const ThreadProfileVisual = ({ type, angle, pitchVal }) => {

    let points = ""; let label = "";

    if (['M', 'UN', 'MF', 'UNC', 'UNF', 'MJ'].includes(type)) {
        const h_thread = p * 0.866; const flat = p / 8;
        points = `M0,${h_box} L0,50 L${p/2-flat},${50+h_thread-flat} L${p/2+flat},${50+h_thread-flat} L${p},50 L${1.5*p-flat},${50+h_thread-flat} L${1.5*p+flat},${50+h_thread-flat} L${2*p},50 L${2.5*p},${50+h_thread} L${2.5*p},${h_box} Z`; label = "60°";
    } else if (['W', 'G', 'R'].includes(type)) {
        const h_thread = p * 0.96;
        points = `M0,${h_box} L0,50 Q${p/4},${50+h_thread/2} ${p/2},${50+h_thread} Q${3*p/4},${50+h_thread/2} ${p},50 Q${1.25*p},${50+h_thread/2} ${1.5*p},${50+h_thread} L${2*p},50 L${2*p},${h_box} Z`; label = "55°";
    } else if (type === 'Tr') {
        const h_thread = p * 0.5; const top = p * 0.366; const slope = (p - top) / 2;
        points = `M0,${h_box} L0,50 L${slope/2},${50+h_thread} L${slope/2+top},${50+h_thread} L${p},50 L${p+slope/2},${50+h_thread} L${p+slope/2+top},${50+h_thread} L${2*p},50 L${2*p},${h_box} Z`; label = "30°";

    }

    return (

            const mn = m; const mt = type === 'helical' ? mn / Math.cos(radBeta) : mn;

            </div>
        </div>

const UnitConverter = () => {
    const [val, setVal] = useState(1);
    const [from, setFrom] = useState('mm');
    const [to, setTo] = useState('inch');
    const res = val * (conversionFactors.length.units[from] / conversionFactors.length.units[to]);
    return (
        <div className="p-6 max-w-2xl mx-auto">
            <div className="bg-white p-6 rounded-xl shadow-sm border border-slate-200">
                <h2 className="text-xl font-bold mb-4 flex gap-2"><Ruler className="text-purple-500"/> Hızlı Çevirici (Uzunluk)</h2>
                <div className="flex gap-4 items-end">
                    <div className="flex-1"><input type="number" value={val} onChange={e=>setVal(e.target.value)} className="input-field"/></div>
                    <select value={from} onChange={e=>setFrom(e.target.value)} className="input-field w-24">{Object.keys(conversionFactors.length.units).map(u=><option key={u}>{u}</option>)}</select>
                    <ArrowRight className="mb-3 text-slate-400"/>
                    <div className="flex-1 text-2xl font-bold text-slate-800 mb-1">{res.toFixed(4)}</div>
                    <select value={to} onChange={e=>setTo(e.target.value)} className="input-field w-24">{Object.keys(conversionFactors.length.units).map(u=><option key={u}>{u}</option>)}</select>
                </div>
            </div>
        </div>
    )
}
